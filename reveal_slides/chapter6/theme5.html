クラスの定義

class Student:
    """学生クラス"""
    
    def __init__(self, name, age, student_id):
        """コンストラクタ（初期化メソッド）"""
        self.name = name        # 属性
        self.age = age          # 属性
        self.student_id = student_id  # 属性
    
    def introduce(self):
        """自己紹介メソッド"""
        return f"私は{self.name}です。{self.age}歳、学籍番号は{self.student_id}です。"

カプセル化

class BankAccount:
    def __init__(self, owner, initial_balance):
        self.owner = owner
        self.__balance = initial_balance  # プライベート属性（__で始まる）
    
    def deposit(self, amount):
        """入金"""
        if amount > 0:
            self.__balance += amount
            return True
        return False
    
    def withdraw(self, amount):
        """出金"""
        if 0 < amount <= self.__balance:
            self.__balance -= amount
            return True
        return False
    
    def get_balance(self):
        """残高の取得"""
        return self.__balance

継承 (Inheritance)

既存のクラスの属性とメソッドを引き継いで新しいクラスを作成する仕組み。

継承の実装例

class Animal:
    """動物クラス（基本クラス）"""
    
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def speak(self):
        return "何らかの音"

class Dog(Animal):
    """犬クラス（派生クラス）"""
    
    def __init__(self, name, age, breed):
        super().__init__(name, age)  # 親クラスの初期化
        self.breed = breed
    
    def speak(self):
        return "ワンワン"
    
    def fetch(self):
        return f"{self.name}がボールを取ってきました"

ポリモーフィズム

class Cat(Animal):
    def speak(self):

class Bird(Animal):

animals = [
    Dog("ポチ", 3, "柴犬"),

ポリモーフィズム (Polymorphism)

同じインターフェースを持つ異なるオブジェクトが、それぞれ独自の動作をする仕組み。

クラス変数とインスタンス変数

class Student:
    school_name = "〇〇高校"  # クラス変数（すべてのインスタンスで共有）
    student_count = 0
    
    def __init__(self, name):
        self.name = name  # インスタンス変数
        Student.student_count += 1  # クラス変数を更新
    
    @classmethod
    def get_student_count(cls):
        return cls.student_count

# 使用例
student1 = Student("太郎")
student2 = Student("花子")

print(Student.school_name)  # "〇〇高校"
print(Student.get_student_count())  # 2

特殊メソッド

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __str__(self):
        """文字列表現"""
        return f"Point({self.x}, {self.y})"
    
    def __eq__(self, other):
        """等価判定"""
        return self.x == other.x and self.y == other.y
    
    def __add__(self, other):
        """加算演算"""
        return Point(self.x + other.x, self.y + other.y)

# 使用例
p1 = Point(1, 2)
p2 = Point(3, 4)

print(p1)           # "Point(1, 2)"
print(p1 + p2)      # "Point(4, 6)"
print(p1 == p2)     # False

実用的な例: 図書管理システム

class Book:
    def __init__(self, title, author, isbn):
        self.title = title
        self.author = author
        self.isbn = isbn
        self.is_borrowed = False
    
    def borrow(self):
        if not self.is_borrowed:
            self.is_borrowed = True
            return True
        return False
    
    def return_book(self):
        if self.is_borrowed:
            self.is_borrowed = False
            return True
        return False

class Library:
    def __init__(self, name):
        self.name = name
        self.books = []
    
    def add_book(self, book):
        self.books.append(book)
    
    def find_book(self, title):
        for book in self.books:
            if book.title == title:
                return book
        return None
    
    def borrow_book(self, title):
        book = self.find_book(title)
        if book and book.borrow():
            return f"'{title}'を貸し出しました"
        return f"'{title}'は貸し出しできません"
