import matplotlib.pyplot as plt

# データの準備
subjects = ['数学', '英語', '国語', '理科', '社会']

# 棒グラフの作成

plt.ylabel('点数', fontsize=12)
plt.ylim(0, 100)

グラフのスタイリング

plt.grid(axis='y', alpha=0.3)

# グラフの表示
plt.tight_layout()
plt.show()

折れ線グラフの作成

import matplotlib.pyplot as plt

# 月別売上データ
months = ['1月', '2月', '3月', '4月', '5月', '6月']
sales = [120, 135, 145, 158, 162, 175]

# 折れ線グラフの作成
plt.figure(figsize=(10, 6))
plt.plot(months, sales, marker='o', linewidth=2, markersize=8)
plt.title('月別売上推移', fontsize=16)
plt.xlabel('月', fontsize=12)
plt.ylabel('売上（万円）', fontsize=12)
plt.grid(True, alpha=0.3)

# データポイントに値を表示
for i, value in enumerate(sales):
    plt.annotate(f'{value}万円', (i, value), 
                textcoords="offset points", xytext=(0,10), ha='center')

plt.tight_layout()
plt.show()

円グラフの作成

import matplotlib.pyplot as plt

# データの準備
categories = ['食費', '住居費', '交通費', '教育費', 'その他']
amounts = [30, 40, 10, 15, 5]
colors = ['#ff9999', '#66b3ff', '#99ff99', '#ffcc99', '#ff99cc']

# 円グラフの作成
plt.figure(figsize=(8, 8))
plt.pie(amounts, labels=categories, colors=colors, autopct='%1.1f%%', 
        startangle=90, textprops={'fontsize': 12})
plt.title('家計支出の割合', fontsize=16)
plt.axis('equal')  # 円を正円にする
plt.show()

散布図の作成

import matplotlib.pyplot as plt
import numpy as np

# データの準備（身長と体重の例）
np.random.seed(42)
height = np.random.normal(170, 10, 50)  # 身長（平均170cm）
weight = 0.8 * height + np.random.normal(0, 5, 50) - 80  # 体重

# 散布図の作成
plt.figure(figsize=(10, 8))
plt.scatter(height, weight, alpha=0.6, s=50)
plt.title('身長と体重の関係', fontsize=16)
plt.xlabel('身長 (cm)', fontsize=12)
plt.ylabel('体重 (kg)', fontsize=12)
plt.grid(True, alpha=0.3)

# 近似直線の追加
z = np.polyfit(height, weight, 1)
p = np.poly1d(z)
plt.plot(height, p(height), "r--", alpha=0.8)

plt.tight_layout()
plt.show()

複数グラフの配置

import matplotlib.pyplot as plt
import numpy as np

# データの準備
x = np.linspace(0, 10, 100)
y1, y2, y3 = np.sin(x), np.cos(x), np.tan(x)

# 2x2のサブプロット
fig, axes = plt.subplots(2, 2, figsize=(12, 10))

サブプロットの設定

# 各グラフの設定
axes[0, 0].plot(x, y1, 'b-')
axes[0, 0].set_title('sin(x)')
axes[0, 0].grid(True)

axes[0, 1].plot(x, y2, 'r-')
axes[0, 1].set_title('cos(x)')
axes[0, 1].grid(True)

axes[1, 0].plot(x, y3, 'g-')
axes[1, 0].set_title('tan(x)')
axes[1, 0].set_ylim(-5, 5)
axes[1, 0].grid(True)

axes[1, 1].text(0.5, 0.5, '予備', ha='center', va='center')

plt.tight_layout()
plt.show()

データ分析と可視化の統合

import pandas as pd
import matplotlib.pyplot as plt

# サンプルデータの作成
data = {
    '月': ['1月', '2月', '3月', '4月', '5月', '6月'],
    '売上': [120, 135, 145, 158, 162, 175],
    '利益': [30, 35, 40, 45, 48, 52]
}

df = pd.DataFrame(data)

複数系列棒グラフの作成

# DataFrameから直接グラフを作成
fig, ax = plt.subplots(figsize=(10, 6))

# 複数系列の棒グラフ
x = range(len(df))
width = 0.35

ax.bar([i - width/2 for i in x], df['売上'], width, 
       label='売上', alpha=0.8)
ax.bar([i + width/2 for i in x], df['利益'], width, 
       label='利益', alpha=0.8)

ax.set_xlabel('月')
ax.set_ylabel('金額（万円）')
ax.set_title('月別売上・利益推移')
ax.set_xticks(x)
ax.set_xticklabels(df['月'])
ax.legend()
ax.grid(axis='y', alpha=0.3)

plt.tight_layout()
plt.show()

インタラクティブな可視化

import tkinter as tk
from tkinter import ttk
import matplotlib.pyplot as plt
from matplotlib.backends.backend_tkagg import FigureCanvasTkAgg
import numpy as np

class GraphApp:
    def __init__(self, root):
        self.root = root
        self.root.title("データ可視化アプリ")
        self.create_widgets()
        self.update_graph()

GUIウィジェットの作成

    def create_widgets(self):
        # コントロールフレーム
        control_frame = ttk.Frame(self.root)
        control_frame.pack(side=tk.TOP, fill=tk.X, padx=5, pady=5)
        
        # グラフ選択
        ttk.Label(control_frame, text="グラフタイプ:").pack(side=tk.LEFT)
        self.graph_type = ttk.Combobox(control_frame, 
                                     values=['sin', 'cos', 'tan'])
        self.graph_type.pack(side=tk.LEFT, padx=5)
        self.graph_type.set('sin')
        
        # 更新ボタン
        update_btn = ttk.Button(control_frame, text="更新", 
                               command=self.update_graph)
        update_btn.pack(side=tk.LEFT, padx=5)

グラフ更新機能

        # グラフ領域の作成
        self.fig, self.ax = plt.subplots(figsize=(8, 6))
        self.canvas = FigureCanvasTkAgg(self.fig, self.root)
        self.canvas.get_tk_widget().pack(fill=tk.BOTH, expand=True)
    
    def update_graph(self):
        self.ax.clear()
        x = np.linspace(0, 2*np.pi, 100)
        
        graph_type = self.graph_type.get()
        if graph_type == 'sin':
            y = np.sin(x)
        elif graph_type == 'cos':
            y = np.cos(x)
        else:
            y = np.clip(np.tan(x), -5, 5)
        
        self.ax.plot(x, y)
        self.ax.set_title(f'{graph_type}(x)')
        self.ax.grid(True)
        self.canvas.draw()

高度な可視化技術

import matplotlib.pyplot as plt
import numpy as np

# データの準備（相関行列の例）
data = np.random.rand(5, 5)
labels = ['A', 'B', 'C', 'D', 'E']

# ヒートマップの作成
fig, ax = plt.subplots(figsize=(8, 6))
im = ax.imshow(data, cmap='coolwarm')

# 軸の設定
ax.set_xticks(np.arange(len(labels)))
ax.set_yticks(np.arange(len(labels)))
ax.set_xticklabels(labels)
ax.set_yticklabels(labels)

# 値の表示
for i in range(len(labels)):
    for j in range(len(labels)):
        text = ax.text(j, i, f'{data[i, j]:.2f}',
                      ha="center", va="center", color="black")

ax.set_title("相関行列")
plt.colorbar(im)
plt.tight_layout()
plt.show()

ヒートマップ (Heatmap)

データの値を色の濃淡で表現する可視化手法。相関行列などでよく使用される。

ダッシュボード (Dashboard)

複数の重要な情報やグラフを一画面にまとめた表示画面。

ダッシュボード作成例

# 学校の成績分析ダッシュボード
fig, axes = plt.subplots(2, 2, figsize=(15, 12))

# 1. 科目別平均点
subject_means = df.mean()
axes[0, 0].bar(subjects, subject_means, color='skyblue')
axes[0, 0].set_title('科目別平均点')
axes[0, 0].set_ylabel('平均点')
axes[0, 0].set_ylim(0, 100)

# 2. 成績分布（ヒストグラム）
all_scores = df.values.flatten()
axes[0, 1].hist(all_scores, bins=10, color='lightgreen', alpha=0.7)
axes[0, 1].set_title('成績分布')
axes[0, 1].set_xlabel('点数')
axes[0, 1].set_ylabel('頻度')

ダッシュボードの完成

# 3. 上位5名の成績
df['総合'] = df.mean(axis=1)
top5 = df.nlargest(5, '総合')
axes[1, 0].barh(top5.index, top5['総合'], color='gold')
axes[1, 0].set_title('総合成績上位5名')
axes[1, 0].set_xlabel('総合平均')

# 4. 科目間相関
correlation = df[subjects].corr()
im = axes[1, 1].imshow(correlation, cmap='coolwarm', vmin=-1, vmax=1)
axes[1, 1].set_xticks(range(len(subjects)))
axes[1, 1].set_yticks(range(len(subjects)))
axes[1, 1].set_xticklabels(subjects)
axes[1, 1].set_yticklabels(subjects)
axes[1, 1].set_title('科目間相関')

plt.tight_layout()
plt.show()
