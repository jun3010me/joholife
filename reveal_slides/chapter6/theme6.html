import csv

def read_csv_file(filename):
    data = []
    with open(filename, 'r', encoding='utf-8') as file:
        csv_reader = csv.reader(file)
        header = next(csv_reader)
        for row in csv_reader:
            data.append(row)
    return header, data

# 使用例
header, data = read_csv_file('population.csv')

データ表示GUI の作成

import tkinter as tk
from tkinter import ttk
import pandas as pd

class DataViewerApp:
    def __init__(self, root):
        self.root = root
        self.root.title("データビューア")
        
        # データフレームの作成
        self.df = pd.DataFrame({
            '商品名': ['りんご', 'バナナ', 'オレンジ'],
            '価格': [100, 80, 120],
            '在庫': [50, 30, 25]
        })
        
        self.create_widgets()
    
    def create_widgets(self):
        # ツリービューウィジェット（表の表示）
        tree = ttk.Treeview(self.root, columns=list(self.df.columns), show='headings')
        
        # ヘッダーの設定
        for col in self.df.columns:
            tree.heading(col, text=col)
            tree.column(col, width=100)
        
        # データの挿入
        for _, row in self.df.iterrows():
            tree.insert('', tk.END, values=list(row))
        
        tree.pack(expand=True, fill='both', padx=10, pady=10)

ファイル操作とGUI

import tkinter as tk
from tkinter import filedialog, messagebox
import pandas as pd

def open_csv_file():
    # ファイル選択ダイアログ
    file_path = filedialog.askopenfilename(
        title="CSVファイルを選択",
        filetypes=[("CSV files", "*.csv"), ("All files", "*.*")]
    )
    
    if file_path:
        try:
            # CSVファイルの読み込み
            df = pd.read_csv(file_path)
            
            # 結果の表示
            result_text = f"読み込み成功！\n"
            result_text += f"行数: {len(df)}\n"
            result_text += f"列数: {len(df.columns)}\n"
            result_text += f"列名: {', '.join(df.columns)}"
            
            messagebox.showinfo("読み込み結果", result_text)
            
        except Exception as e:
            messagebox.showerror("エラー", f"ファイル読み込みエラー: {e}")

# GUI要素の作成
root = tk.Tk()
root.title("CSVファイル読み込みアプリ")

button = tk.Button(root, text="CSVファイルを開く", command=open_csv_file)
button.pack(pady=20)

root.mainloop()

import pandas as pd

def process_sales_data(filename):
    """売上データの処理"""
    try:
        # データの読み込み
        df = pd.read_csv(filename)
        
        # 日付列の変換
        df['日付'] = pd.to_datetime(df['日付'])
        
        # 月別売上の集計
        monthly_sales = df.groupby(df['日付'].dt.month)['売上'].sum()
        
        # 商品別売上の集計
        product_sales = df.groupby('商品名')['売上'].sum()
        
        return monthly_sales, product_sales
        
    except Exception as e:
        print(f"データ処理エラー: {e}")
        return None, None

実用的なデータ活用例

class GradeManager:
    def __init__(self):
        self.df = pd.DataFrame(columns=[
            '学籍番号', '名前', '数学', '英語', '国語', '平均'
        ])
    
    def add_student(self, student_id, name, math, eng, jpn):
        """学生データの追加"""
        average = (math + eng + jpn) / 3
        new_row = {
            '学籍番号': student_id, '名前': name,
            '数学': math, '英語': eng, '国語': jpn,
            '平均': round(average, 1)
        }
        self.df = pd.concat([self.df, pd.DataFrame([new_row])], 
                           ignore_index=True)

成績管理システムの機能

    def get_class_average(self):
        """クラス平均の計算"""
        return {
            '数学': self.df['数学'].mean(),
            '英語': self.df['英語'].mean(),
            '国語': self.df['国語'].mean(),
            '総合': self.df['平均'].mean()
        }
    
    def save_to_csv(self, filename):
        """ＣＳＶファイルに保存"""
        self.df.to_csv(filename, index=False, encoding='utf-8')

# JSONファイルとの連携
import json

def save_config(config, filename):
    """設定をJSONファイルに保存"""
    with open(filename, 'w', encoding='utf-8') as f:
        json.dump(config, f, ensure_ascii=False, indent=2)

def load_config(filename):
    """JSONファイルから設定を読み込み"""
    with open(filename, 'r', encoding='utf-8') as f:
        return json.load(f)

# 使用例
save_config(app_config, 'config.json')
loaded_config = load_config('config.json')
